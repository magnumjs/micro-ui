createState(initialState) State

Creates a reactive state container. 

const state = createState({ count: 0 });

state.subscribe(newState => {
  console.log(newState.count);
});

state.setState({ count: 1 });

createComponent(renderFn) View

Declares a UI component from a render function that receives props. 

const Counter = createComponent(({ count }) => `
  <h1>Count: ${count}</h1>
  <button id="inc">+</button>
`);

Counter.mount(document.getElementById('app'));
Counter.update({ count: 0 });
